
#version 460 core

#ifdef GL_ES
precision mediump float;
#endif

uniform vec2 u_mouse;
uniform vec2 u_resolution;

out vec4 FragColor;

void main()
{
    vec2 normCoords = gl_FragCoord.xy/(u_resolution); // normalized resolution coordinates
    FragColor = vec4(normCoords.x, 0.0, normCoords.y, 1.0f);
}
